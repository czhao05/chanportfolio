SELECT name, continent, inflation_rate
  FROM countries
  INNER JOIN economies
  USING(code)
  WHERE year = 2015
    AND inflation_rate IN (
        SELECT MAX(inflation_rate) AS max_inf
        FROM (
             SELECT name, continent, inflation_rate
             FROM countries
             INNER JOIN economies
             USING(code)
             WHERE year = 2015) AS subquery
        GROUP BY continent);
